const audio = document.getElementById('invite-audio'); const musicBtn = document.getElementById('music-btn'); const iconPlay = document.querySelector('.icon-play'); const iconPause = document.querySelector('.icon-pause'); musicBtn.addEventListener('click', () => { if (audio.paused) { // Включаем воспроизведение audio.play().then(() => { musicBtn.classList.add('playing'); iconPlay.style.display = 'none'; // Скрываем "Play" iconPause.style.display = 'block'; // Показываем "Pause" }).catch(error => { console.log("Ошибка воспроизведения аудио:", error); }); } else { // Ставим на паузу audio.pause(); musicBtn.classList.remove('playing'); iconPlay.style.display = 'block'; // Показываем "Play" iconPause.style.display = 'none'; // Скрываем "Pause" } });
